#### 基本用法：

```tsx
import { Slider } from "cola-react-ui";

const MySelect: React.FC = () => {
  const onChange = (value: number) => {
    console.log("onChange: ", value);
  };
  const onAfterChange = (value: number) => {
    console.log("onAfterChange: ", value);
  };
  return (
    <Slider
      style={{ width: 400 }}
      defaultValue={50}
      max={100}
      min={0}
      onChange={onChange}
      onAfterChange={onChange}
    />
  );
};
```
