#### 引入：
```tsx

```
#### 基本用法：
```tsx
import { Input } from 'cola-react-ui'

const MyButton: React.FC = () => {
  const handleChange = (e) => {
    console.log(e.target.value)
  }
  return(
    <>
      <Input
        placeholder='input'
        style={{width: 300}}
        onChange={handleChange}
      />
      <Input.Password placeholder='password' style={{width: 300}}/>
    </>
  )
}
```
