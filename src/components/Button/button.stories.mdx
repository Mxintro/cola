#### 引入：
```tsx
  import { Button } from 'cola-react-ui'
```
#### 基本用法：
```tsx
import { Button } from 'cola-react-ui'

const MyButton: React.FC = () => {
  const handleClick = () => {
    console.log('do something')
  }
  return (
    <Button btnType='primary'onClick={handleClick}> 
      button
    </Button>
  )
}
```
